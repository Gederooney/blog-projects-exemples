<div class="relative">
  <input 
	type="text" 
	name="username" 
	id="username" 
	placeholder=""
  />
  <label 
  	for="username"
  >
  	Username
  </label>
</div>



.relative {
  position: relative;
}
input {
  padding: 1rem;
  border: 2px solid #f1b4bb;
  border-radius: 0.5rem;
  background-color: transparent;
  color: #fff;
  font-size: 0.75rem;
  font-weight: 300;
  letter-spacing: 0.05rem;
  outline: none;
}
label {
  position: absolute;
  top: 50%;
  transform: translateY(-50%);
  background-color: transparent;
  transition: all 0.2s 
   cubic-bezier(0.895, 0.03, 0.685, 0.22);
  color: #fff;
  font-size: 0.75rem;
  font-weight: 300;
  letter-spacing: 0.05rem;
  pointer-events: none;
}
input:focus ~ label,
input:not(:placeholder-shown) ~ label {
  top: 0;
  left: 1rem;
  background-color: #1f4172;
}
